import java.util.*;
public class Main
{
	public static void main(String[] args) {
	    Scanner sc = new Scanner(System.in);
	    System.out.println("Please input r and i");
	    int r = sc.nextInt();
	    int i = sc.nextInt();
	    Complex c = new Complex();
	    c.setR(r);
	    c.setI(i);
	    c.print();
	  System.out.println("Thanks for using me!"); 

	}
}

 class Complex{
    int r= 9, i = 10;
    void print()
    {
        System.out.println(r+ " +i"+ i);
    }
    void setR(int a){
        r = a;
    }
    void setI(int a){
        i = a;
    }
}
